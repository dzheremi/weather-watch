    class StationFeed

    This class is used to produce a JSON feed for a set of stations, when given a groups of states. The class does not inherit from ActiveRecord, all attributes are destroyed when the web session completes.

def set_stations(state)

  data = []
  state.each do |s|
    stations = []
    s.stations.each do |st|
      stations << {
          station_name: st.name,
          station_id: st.id,
          station_lat: st.latitude,
          station_lon: st.longitude,
          station_tz: st.timezone.name
      }
    end
    data << {
        name: s.name,
        stations: stations
    }
  end

  self.states = data

end
